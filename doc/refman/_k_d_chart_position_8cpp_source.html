203 QList<QByteArray> Position::names( Options options )

205  QList<QByteArray> list;
206  const int start = ( options & IncludeCenter ) ? 1 : 2;
207  const int end = ( options & IncludeFloating ) ? maxPositionValue : maxPositionValue-1;
